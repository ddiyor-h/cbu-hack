X_test = pd.read_parquet('X_test.parquet')

# –£–¥–∞–ª–∏—Ç—å ID-–ø—Ä–∏–∑–Ω–∞–∫–∏

print(f"–ì–æ—Ç–æ–≤–æ! –ü—Ä–∏–∑–Ω–∞–∫–æ–≤ –¥–ª—è –æ–±—É—á–µ–Ω–∏—è: {X_train_clean.shape[1]}")

### 2. –û–±—É—á–∏—Ç—å –±–∞–∑–æ–≤—É—é –º–æ–¥–µ–ª—å

```python
import lightgbm as lgb
from sklearn.metrics import roc_auc_score

# –°–æ–∑–¥–∞—Ç—å –º–æ–¥–µ–ª—å

    n_estimators=1000,
    learning_rate=0.05,
    max_depth=7,
    num_leaves=31,
    random_state=42,
    n_jobs=-1
)

# –û–±—É—á–∏—Ç—å
model.fit(
    X_train_clean, y_train.values.ravel(),
    eval_set=[(X_test_clean, y_test.values.ravel())],
    eval_metric='auc',
    early_stopping_rounds=50,
    verbose=100
)

# –ü—Ä–µ–¥—Å–∫–∞–∑–∞—Ç—å
y_pred_proba = model.predict_proba(X_test_clean)[:, 1]

# –û—Ü–µ–Ω–∏—Ç—å AUC

print(f"\nTest AUC: {auc:.4f}")
```
